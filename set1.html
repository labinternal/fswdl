Question 1: JavaScript function for vowels in a string

<!DOCTYPE HTML>
<html>
<head>
<script type="text/javascript">
function str_vowel() {
  var str = prompt("Enter a string", "");
  for (var i = 0; i < str.length; i++) {
    if ("aeiouAEIOU".includes(str.charAt(i))) {
      document.write("The entered string is: " + str + "<br/>");
      document.write("The leftmost vowel is: " + str.charAt(i) + "<br/>");
      document.write("The position of the leftmost vowel is: " + (i + 1));
      return;
    }
  }
  document.write("The entered string is: " + str + "<br/>");
  document.write("The entered string has no vowels");
}
</script>
</head>
<body onload="str_vowel()">
</body>
</html>

Question 2: jQuery get() using XAMPP (Task5.html)

<html lang="en">

  <meta charset="utf-8">
  <title>jQuery get() Demo</title>
  <script src="https://code.jquery.com/jquery-3.5.1.min.js"></script>
  <script>
    $(document).ready(function(){
      $("button").click(function(){
        $.get("date.php", function(data){
          $("#result").html(data);
        });
      });
    });
  </script>
</head>
<body>
  <div id="result">
    <h2>The XMLHttpRequest Object by the server date and time</h2>

  </div>
