Task 1: JavaScript Functions

Task 1a: Find Left-Most Vowel Position

<!DOCTYPE html>
<html>
<head>
    <script>
    function findVowelPosition() {
        const str = prompt("Enter a string:");
        const vowels = ['a', 'e', 'i', 'o', 'u'];
        
        for(let i = 0; i < str.length; i++) {
            if(vowels.includes(str[i].toLowerCase())) {
                document.getElementById("output").innerHTML = `
                    String: ${str}<br>
                    Left-most vowel: '${str[i]}'<br>
                    Position: ${i + 1}
                `;
                return;
            }
        }
        document.getElementById("output").innerHTML = "No vowels found";
    }
    </script>
</head>
<body>
    <button onclick="findVowelPosition()">Find Vowel Position</button>
    <div id="output"></div>
</body>
</html>

Task 1b: Reverse Number Digits

<!DOCTYPE html>
<html>
<head>
    <script>
    function reverseNumber() {
        let num = parseInt(prompt("Enter a number:"));
        let reversed = 0;
        
        while(num > 0) {
            reversed = reversed * 10 + (num % 10);
            num = Math.floor(num / 10);
        }
        
        document.getElementById("output").innerHTML = `Reversed: ${reversed}`;
    }
    </script>
</head>
<body>
    <button onclick="reverseNumber()">Reverse Number</button>
    <div id="output"></div>
</body>
</html>

How to Execute:

  1. Copy the code for either Task 1a or 1b
  2. Save as an HTML file (e.g., vowel.html or reverse.html)
  3. Open the file in any web browser
  4. Click the button and follow the prompts
