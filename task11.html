Task 11: PHP States Program Code

<?php
$states = "Mississippi Alabama Texas Massachusetts Kansas";
$state_array = explode(' ', $states);
$result = array();

foreach($state_array as $state) {
    $len = strlen($state);
    
    // a. Ends with xas
    if(substr_compare($state, "xas", $len-3, 3) === 0) {
        $result[0] = $state;
    }
    
    // b. Starts with K/k and ends with s
    if(strtolower($state[0]) == 'k' && strtolower($state[$len-1]) == 's') {
        $result[1] = $state;
    }
    
    // c. Starts with M and ends with s
    if($state[0] == 'M' && $state[$len-1] == 's') {
        $result[2] = $state;
    }
    
    // d. Ends with a
    if($state[$len-1] == 'a') {
        $result[3] = $state;
    }
}

echo "<pre>";
print_r($result);
echo "</pre>";
?>
