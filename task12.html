Task 12: PHP Student Sorting

<?php
$conn = mysqli_connect("localhost","root","","student");
$sql = "SELECT * FROM studentinfo";
$result = mysqli_query($conn, $sql);
$students = array();

while($row = mysqli_fetch_assoc($result)) {
    $students[$row['usn']] = $row;
}

// Selection sort
$usns = array_keys($students);
for($i=0;$i<count($usns)-1;$i++) {
    $min_idx = $i;
    for($j=$i+1;$j<count($usns);$j++) {
        if($usns[$j] < $usns[$min_idx]) $min_idx = $j;
    }
    $temp = $usns[$i];
    $usns[$i] = $usns[$min_idx];
    $usns[$min_idx] = $temp;
}

echo "<table border='1'>
    <tr><th>USN</th><th>Name</th><th>Address</th></tr>";
foreach($usns as $usn) {
    echo "<tr>
        <td>".$students[$usn]['usn']."</td>
        <td>".$students[$usn]['name']."</td>
        <td>".$students[$usn]['address']."</td>
    </tr>";
}
echo "</table>";
?>
