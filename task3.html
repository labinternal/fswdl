Task 3: Text Animation

Working Demo:

Complete Code:

<!DOCTYPE html>
<html>
<head>
    <title>Text Growing/Shrinking</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            text-align: center;
            margin-top: 50px;
        }
        h1 {
            transition: font-size 0.1s;
        }
    </style>
</head>
<body>
    <div id="text-container"></div>
    
    <script>
        let size = 5;
        let growing = true;
        let text = "TEXT-GROWING";
        const container = document.getElementById('text-container');
        
        function animateText() {
            if (growing) {
                size += 5;
                container.innerHTML = `<h1 style="color: red;">${text}</h1>`;
                
                if (size >= 50) {
                    growing = false;
                    text = "TEXT-SHRINKING";
                }
            } else {
                size -= 5;
                container.innerHTML = `<h1 style="color: blue;">${text}</h1>`;
                
                if (size <= 5) {
                    growing = true;
                    text = "TEXT-GROWING";
                }
            }
            
            container.firstChild.style.fontSize = `${size}pt`;
            setTimeout(animateText, 100);
        }
        
        animateText();
    </script>
</body>
</html>

How to Execute:

  1. Copy the entire code block above
  2. Save as an HTML file (e.g., text-animation.html)
  3. Open the file in any web browser
  4. The animation will start automatically
