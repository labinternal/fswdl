Q1) Growing and Shrinking Text

  <!DOCTYPE html>
  <html>
  <body>
  <div id="h"></div>
  <script>
  var v = 0, f = 1, t = "TEXT-GROWING", color;

  function a() {
      if (f == 1) {
          v += 5;
          color = "red";
      } else {
          v -= 5;
          color = "blue";
      }

      document.getElementById("h").innerHTML =
          "<h1 style='font-size:" + v + "px; margin:0px; color:" + color + "'><b>" + t + "</b></h1>";

      if (v == 50) {
          f = 0;
          t = "TEXT-SHRINKING";
      }
      if (v == 5) {
          f = 1;
          t = "TEXT-GROWING";
      }

      setTimeout(a, 100);
  }

  a();
  </script>
  </body>
  </html>
